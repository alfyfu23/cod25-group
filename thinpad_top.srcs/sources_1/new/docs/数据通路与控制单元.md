# RV32I CPU 数据通路与控制单元详解

## 1. 数据通路概述

数据通路是CPU中处理和传输数据的核心路径，包括取指、译码、执行、访存和写回五个阶段的数据流动。控制单元负责生成控制信号，指导数据在通路中的正确流动和操作。

## 2. 取指阶段数据通路

### 2.1 PC生成与更新

PC（程序计数器）是取指阶段的核心，负责提供当前指令地址：

```systemverilog
logic [31:0] pc_reg;  // PC寄存器

// PC更新逻辑
always_ff @(posedge clk or posedge rst) begin
    if (rst) begin
        pc_reg <= RESET_VECTOR;
    end else if (trap_flush) begin
        pc_reg <= trap_target_pc_reg;
    end else if (branch_flush) begin
        if (branch_taken_ex)
            pc_reg <= branch_target_ex;
        else
            pc_reg <= id_ex_reg.pc + 32'd4;
    end else if (fencei_flush) begin
        pc_reg <= ex_mem_reg.pc_plus4;
    end else if (mret_flush) begin
        pc_reg <= csr_mepc;
    end else if (icache_hit) begin
        if (btb_pred_taken)
            pc_reg <= btb_pred_target;
        else
            pc_reg <= pc_reg + 32'd4;
    end
end
```

### 2.2 指令缓存(I-Cache)

I-Cache采用直接映射结构，提高指令获取效率：

```systemverilog
localparam ICACHE_INDEX_BITS = 7;  // 128行
localparam ICACHE_TAG_BITS = 32 - 2 - ICACHE_INDEX_BITS;

logic [31:0] icache_data [0:(1<<ICACHE_INDEX_BITS)-1];
logic [ICACHE_TAG_BITS-1:0] icache_tag [0:(1<<ICACHE_INDEX_BITS)-1];
logic icache_valid [0:(1<<ICACHE_INDEX_BITS)-1];

// 缓存命中判断
assign icache_index   = pc_reg[ICACHE_INDEX_BITS+1:2];
assign icache_req_tag = pc_reg[31:ICACHE_INDEX_BITS+2];
assign icache_hit     = icache_valid[icache_index] && (icache_tag[icache_index] == icache_req_tag);
```

### 2.3 分支目标缓冲器(BTB)

BTB用于分支预测，减少分支指令导致的性能损失：

```systemverilog
localparam BTB_SIZE = 64;
localparam BTB_INDEX_BITS = 6;
localparam BTB_TAG_BITS = 32 - 2 - BTB_INDEX_BITS;

typedef struct packed {
    logic        valid;
    logic [BTB_TAG_BITS-1:0] tag;
    logic [31:0] target;
    logic [1:0]  counter;  // 2位饱和计数器
} btb_entry_t;

btb_entry_t btb [BTB_SIZE-1:0];

// 分支预测逻辑
assign btb_hit = btb[if_btb_index].valid && (btb[if_btb_index].tag == if_btb_tag);
assign btb_pred_taken = btb_hit && btb[if_btb_index].counter[1];
assign btb_pred_target = btb[if_btb_index].target;
```

## 3. 译码阶段数据通路

### 3.1 寄存器堆

寄存器堆提供32个32位通用寄存器，支持双端口读取和单端口写入：

```systemverilog
module rv32i_regfile (
    input  wire        clk,
    input  wire        rst,
    input  wire        we,           // 写使能
    input  wire [4:0]  waddr,        // 写地址
    input  wire [31:0] wdata,        // 写数据
    input  wire [4:0]  raddr1, raddr2,  // 读地址
    output logic [31:0] rdata1, rdata2   // 读数据
);

logic [31:0] regs[31:0];

// 写操作
always_ff @(posedge clk or posedge rst) begin
    if (rst) begin
        for (int i = 0; i < 32; i++)
            regs[i] <= 32'h0;
    end else if (we && (waddr != 5'd0)) begin
        regs[waddr] <= wdata;
    end
end

// 读操作
always_comb begin
    rdata1 = (raddr1 == 5'd0) ? 32'h0 : regs[raddr1];
    rdata2 = (raddr2 == 5'd0) ? 32'h0 : regs[raddr2];
end
```

### 3.2 立即数生成器

根据指令类型生成不同格式的立即数：

```systemverilog
// I型立即数
function automatic logic [31:0] imm_i(input logic [31:0] instr);
    imm_i = {{20{instr[31]}}, instr[31:20]};
endfunction

// S型立即数
function automatic logic [31:0] imm_s(input logic [31:0] instr);
    imm_s = {{20{instr[31]}}, instr[31:25], instr[11:7]};
endfunction

// B型立即数
function automatic logic [31:0] imm_b(input logic [31:0] instr);
    imm_b = {{19{instr[31]}}, instr[31], instr[7], instr[30:25], instr[11:8], 1'b0};
endfunction

// U型立即数
function automatic logic [31:0] imm_u(input logic [31:0] instr);
    imm_u = {instr[31:12], 12'h000};
endfunction

// J型立即数
function automatic logic [31:0] imm_j(input logic [31:0] instr);
    imm_j = {{11{instr[31]}}, instr[31], instr[19:12], instr[20], instr[30:21], 1'b0};
endfunction
```

### 3.3 控制单元

控制单元根据指令 opcode、funct3 和 funct7 字段生成各类控制信号：

```systemverilog
always_comb begin
    // 默认值
    id_reg_write   = 1'b0;
    id_mem_read    = 1'b0;
    id_mem_write   = 1'b0;
    id_mem_size    = MEM_SIZE_WORD;
    id_mem_unsigned= 1'b0;
    id_op_a_sel    = OP_A_RS1;
    id_op_b_sel    = OP_B_RS2;
    id_alu_op      = ALU_ADD;
    id_branch      = BR_NONE;
    id_jump        = 1'b0;
    id_jalr        = 1'b0;
    id_wb_src      = WB_SRC_ALU;
    id_csr_en      = 1'b0;
    id_illegal     = id_valid;
    id_use_rs1     = 1'b0;
    id_use_rs2     = 1'b0;
    
    if (id_valid) begin
        unique case (id_opcode)
            OPCODE_LUI: begin
                id_reg_write = 1'b1;
                id_op_a_sel  = OP_A_ZERO;
                id_op_b_sel  = OP_B_IMM;
                id_alu_op    = ALU_PASS_B;
                id_imm_value = imm_u(id_instr);
                id_illegal   = 1'b0;
            end
            OPCODE_AUIPC: begin
                id_reg_write = 1'b1;
                id_op_a_sel  = OP_A_PC;
                id_op_b_sel  = OP_B_IMM;
                id_alu_op    = ALU_ADD;
                id_imm_value = imm_u(id_instr);
                id_illegal   = 1'b0;
            end
            OPCODE_JAL: begin
                id_reg_write = 1'b1;
                id_jump      = 1'b1;
                id_wb_src    = WB_SRC_PC4;
                id_op_a_sel  = OP_A_PC;
                id_op_b_sel  = OP_B_IMM;
                id_alu_op    = ALU_ADD;
                id_imm_value = imm_j(id_instr);
                id_illegal   = 1'b0;
            end
            // ... 其他指令类型
        endcase
    end
end
```

## 4. 执行阶段数据通路

### 4.1 前递单元

前递单元解决数据冲突，将后续阶段的计算结果直接传递给执行阶段：

```systemverilog
always_comb begin
    // 默认使用ID阶段读取的寄存器值
    ex_rs1_value = id_ex_reg.rs1_value;
    ex_rs2_value = id_ex_reg.rs2_value;
    
    // EX/MEM阶段前递（优先级高）
    if (id_ex_reg.use_rs1 && ex_mem_reg.valid && ex_mem_reg.reg_write && 
        (ex_mem_reg.rd != 5'd0) && (ex_mem_reg.rd == id_ex_reg.rs1)) begin
        ex_rs1_value = ex_mem_forward_value;
    end
    
    // MEM/WB阶段前递
    if (id_ex_reg.use_rs1 && mem_wb_reg.valid && mem_wb_reg.reg_write && 
        (mem_wb_reg.rd != 5'd0) && (mem_wb_reg.rd == id_ex_reg.rs1)) begin
        ex_rs1_value = mem_wb_forward_value;
    end
    
    // 类似处理rs2...
end
```

### 4.2 ALU（算术逻辑单元）

ALU执行各种算术和逻辑运算：

```systemverilog
always_comb begin
    // 操作数选择
    case (id_ex_reg.op_a_sel)
        OP_A_PC:   alu_operand_a = id_ex_reg.pc;
        OP_A_ZERO: alu_operand_a = 32'h0;
        default:   alu_operand_a = ex_rs1_value;
    endcase

    case (id_ex_reg.op_b_sel)
        OP_B_IMM:  alu_operand_b = id_ex_reg.imm;
        OP_B_FOUR: alu_operand_b = 32'd4;
        default:   alu_operand_b = ex_rs2_value;
    endcase
    
    // ALU运算
    case (id_ex_reg.alu_op)
        ALU_ADD:   alu_result = alu_operand_a + alu_operand_b;
        ALU_SUB:   alu_result = alu_operand_a - alu_operand_b;
        ALU_SLL:   alu_result = alu_operand_a << alu_operand_b[4:0];
        ALU_SLT:   alu_result = ($signed(alu_operand_a) < $signed(alu_operand_b)) ? 32'd1 : 32'd0;
        ALU_SLTU:  alu_result = (alu_operand_a < alu_operand_b) ? 32'd1 : 32'd0;
        ALU_XOR:   alu_result = alu_operand_a ^ alu_operand_b;
        ALU_XNOR:  alu_result = ~(alu_operand_a ^ alu_operand_b);
        ALU_SRL:   alu_result = alu_operand_a >> alu_operand_b[4:0];
        ALU_SRA:   alu_result = $signed(alu_operand_a) >>> alu_operand_b[4:0];
        ALU_OR:    alu_result = alu_operand_a | alu_operand_b;
        ALU_AND:   alu_result = alu_operand_a & alu_operand_b;
        ALU_PASS_B:alu_result = alu_operand_b;
        ALU_MINU:  alu_result = (alu_operand_a < alu_operand_b) ? alu_operand_a : alu_operand_b;
        ALU_CTZ:   alu_result = ctz32(ex_rs1_value);
        default:   alu_result = 32'h0;
    endcase
end
```

### 4.3 分支判断单元

分支判断单元计算分支条件是否满足：

```systemverilog
always_comb begin
    branch_taken_ex = 1'b0;
    branch_target_ex = id_ex_reg.pc + id_ex_reg.imm;

    case (id_ex_reg.branch)
        BR_EQ:  branch_taken_ex = (ex_rs1_value == ex_rs2_value);
        BR_NE:  branch_taken_ex = (ex_rs1_value != ex_rs2_value);
        BR_LT:  branch_taken_ex = ($signed(ex_rs1_value) < $signed(ex_rs2_value));
        BR_GE:  branch_taken_ex = ($signed(ex_rs1_value) >= $signed(ex_rs2_value));
        BR_LTU: branch_taken_ex = (ex_rs1_value < ex_rs2_value);
        BR_GEU: branch_taken_ex = (ex_rs1_value >= ex_rs2_value);
        default: ;
    endcase

    if (id_ex_reg.jump) begin
        branch_taken_ex = 1'b1;
        if (id_ex_reg.jalr)
            branch_target_ex = (ex_rs1_value + id_ex_reg.imm) & 32'hFFFF_FFFE;
        else
            branch_target_ex = id_ex_reg.pc + id_ex_reg.imm;
    end
end
```

## 5. 访存阶段数据通路

### 5.1 数据对齐与字节使能

处理不同宽度的内存访问：

```systemverilog
always_comb begin
    store_sel_comb  = 4'b0000;
    store_data_comb = 32'h0;
    case (id_ex_reg.mem_size)
        MEM_SIZE_BYTE: begin
            store_sel_comb  = 4'b0001 << alu_result[1:0];
            store_data_comb = {4{ex_rs2_value[7:0]}} << (8 * alu_result[1:0]);
        end
        MEM_SIZE_HALF: begin
            store_sel_comb  = alu_result[1] ? 4'b1100 : 4'b0011;
            store_data_comb = {2{ex_rs2_value[15:0]}} << (16 * alu_result[1]);
        end
        MEM_SIZE_WORD: begin
            store_sel_comb  = 4'b1111;
            store_data_comb = ex_rs2_value;
        end
    endcase
end
```

### 5.2 数据加载与符号扩展

处理加载指令的数据格式转换：

```systemverilog
always_comb begin
    logic [7:0]  load_byte;
    logic [15:0] load_half;
    load_byte = mem_transaction_data >> (8 * ex_mem_reg.mem_addr[1:0]);
    load_half = mem_transaction_data >> (16 * ex_mem_reg.mem_addr[1]);
    mem_load_data = mem_transaction_data;
    
    case (ex_mem_reg.mem_size)
        MEM_SIZE_BYTE: begin
            if (ex_mem_reg.mem_unsigned)
                mem_load_data = {24'h0, load_byte};
            else
                mem_load_data = {{24{load_byte[7]}}, load_byte};
        end
        MEM_SIZE_HALF: begin
            if (ex_mem_reg.mem_unsigned)
                mem_load_data = {16'h0, load_half};
            else
                mem_load_data = {{16{load_half[15]}}, load_half};
        end
        default: mem_load_data = mem_transaction_data;
    endcase
end
```

## 7. 控制信号详解
